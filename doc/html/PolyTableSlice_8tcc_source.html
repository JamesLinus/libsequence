81  template<typename T>
82  void PolyTableSlice<T>::process_windows( const PolyTable::const_site_iterator beg,
83  const PolyTable::const_site_iterator end,
84  const unsigned & window_size,
85  const unsigned & step_len,
86  const double & alignment_length,
87  const bool & is_physical,
88  const double & physical_scale )
89  {
90  if (window_size > 0 && step_len > 0 )
91  {
92  if (end>beg)
93  {
94  unsigned nsites = end-beg;
95  if (is_physical)
96  {
97  PolyTable::const_site_iterator b=beg,e=end;
98  if (alignment_length>0. && physical_scale > 0.)
99  {
100  unsigned i = 0;
101  double currLen = double(i*step_len);
102  while(currLen < alignment_length)
103  {
104  double endOfWindow = currLen + double(window_size);
105  unsigned off1=Sequence::SEQMAXUNSIGNED,
106  off2=Sequence::SEQMAXUNSIGNED;
107  size_t j=0;
108  while ((b+j) < e)
109  {
110  if ( physical_scale*(b+j)->first > currLen &&
111  physical_scale*(b+j)->first <= endOfWindow )
112  {
113  if (off1 == Sequence::SEQMAXUNSIGNED)
114  {
115  off1=j;
116  off2=j+1; //we add 1 because off2 will be used to make an "end" iterator
117  }
118  else
119  {
120  ++off2;
121  }
122  ++j;
123  }
124  else ++j;
125  if ( (b+j)>=e || physical_scale*(b+j)->first > endOfWindow )
126  break;
127  }
128  if (off1 != Sequence::SEQMAXUNSIGNED)
129  {
130  windows.push_back( std::make_pair( (b+off1),(b+off2) ) );
131  }
132  else
133  {
134  windows.push_back( std::make_pair( end,end ) );
135  }
136  ++i;
137  currLen = double(i*step_len);
138  }
139  }
140  }
141  else
142  {
143  unsigned jump=0;
144  for (unsigned i=0 ; i<nsites ; i += jump)
145  {
146  //A check is necessary here--
147  //We are trying to guarantee that there are exactly k polymorphisms
148  //in each window. However, SNP tables can be constructed/manipulated
149  //such that each column is not neccessarily polymorphic.
150  jump=0;
151  unsigned k = 0;
152  while(jump < step_len && i+k<nsites)
153  {
154  stateCounter counts = std::for_each( (beg+i+k)->second.begin(),
155  (beg+i+k)->second.end(),
156  stateCounter('-'));
157  if (counts.nStates() > 1) //is a polymorphic site
158  {
159  ++jump;
160  }
161  ++k;
162  }
163  windows.push_back( std::make_pair( (beg+i),(beg+i+k) ) );
164 // if ( (beg+i+step_len+1) < end )
165 // {
166 // windows.push_back( std::make_pair( (beg+i),(beg+i+step_len+1) ) );
167 // }
168 // else
169 // {
170 // windows.push_back( std::make_pair( (beg+i),end ) );
171 // }
172  }
173  }
174  windows_begin = windows.begin();
175  windows_end = windows.end();
176  }
177  }
178  }
