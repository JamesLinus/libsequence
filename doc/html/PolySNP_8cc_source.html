98  void _PolySNPImpl::preprocess()
114  {
115  if (!_preprocessed)
116  {
117  for(unsigned site = 0 ; site < _nsites ; ++site)
118  {
119  for (unsigned seq = 0 ; seq < _nsam ; ++seq)
120  {
121  //process counts w/o respect to
122  //ancestral or derived
123  if (!_haveOutgroup
124  || (_haveOutgroup && seq != _outgroup))
125  {
126  _counts[site]( (*_data)[seq][site] );
127  }
128  //process derived states if outgroup is
129  //present
130  if (_haveOutgroup == true)
131  {
132  //if outgroup state is missing data
133  //or a gap
134  //set the bool for that site to false
135  if ( std::toupper((*_data)[_outgroup][site]) == 'N' ||
136  (*_data)[_outgroup][site] == '-')
137  {
138  _derivedCounts[site].first = false;
139  }
140  else
141  {
142  //tabulate the derived state
143  _derivedCounts[site].first = true;
144  if ( seq != _outgroup && (*_data)[seq][site] != (*_data)[_outgroup][site] )
145  {
146  _derivedCounts[site].second((*_data)[seq][site]);
147  }
148  }
149  }
150  else
151  {
152  //if no outgroup, set bool
153  //for site to false
154  _derivedCounts[site].first = false;
155  }
156  }
157  if(_counts[site].nStates() > 1 && _counts[site].gap==0) ++_NumPoly;
158  }
159  _preprocessed = true;
160  }

227  SSH += (rep->_counts[i].g > 0) ? double(rep->_counts[i].g) *

330  {
331  assert ( rep->_preprocessed );
332  if (rep->_NumPoly==0)
333  return 0.;
334  if( !rep->_haveOutgroup)
335  return std::numeric_limits<double>::quiet_NaN();
336  double H = 0.0;
337  bool anc_is_present = 0; //is ancestral state present in the ingroup?
338 
339  for (unsigned i = 0; i < rep->_nsites; ++i)
340  { //iterate over sites
341  if ( rep->_derivedCounts[i].second.gap == 0)
342  {
343  unsigned samplesize = rep->_totsam; //sample size per site
344  unsigned sumDerCounts = 0;
345  sumDerCounts += rep->_derivedCounts[i].second.a;
346  sumDerCounts += rep->_derivedCounts[i].second.g;
347  sumDerCounts += rep->_derivedCounts[i].second.c;
348  sumDerCounts += rep->_derivedCounts[i].second.t;
349  sumDerCounts += rep->_derivedCounts[i].second.zero;
350  sumDerCounts += rep->_derivedCounts[i].second.one;
351  unsigned ancestralCounts = samplesize-sumDerCounts-rep->_derivedCounts[i].second.n;
352 
353  //if the ancestral state is not missing data, and
354  //the sum of the derived counts + and missing data in the ingroup
355  //does not equal the sample size, then the ancestral state is present
356  //at least once in the ingroup, and anc_is_present = true
357  anc_is_present = (rep->_derivedCounts[i].first == true &&
358  ancestralCounts > 0 ) ? true : false;
359  if (anc_is_present) //ancestral state must be present
360  {
361  //number of derived states seen at this site
362  unsigned numDer = rep->_derivedCounts[i].second.nStates();
363  if (numDer == 1)
364  { //simple if there is only one derived state inferred
365  samplesize -= rep->_derivedCounts[i].second.n;
366  double denom = (double(samplesize) *
367  (double(samplesize) - 1.0));
368  H += (rep->_derivedCounts[i].second.a > 0 ) ?
369  (2.0 * pow (double (rep->_derivedCounts[i].second.a), 2.0)) / denom : 0. ;
370  H += (rep->_derivedCounts[i].second.g > 0 ) ?
371  (2.0 * pow (double (rep->_derivedCounts[i].second.g), 2.0)) / denom : 0. ;
372  H += (rep->_derivedCounts[i].second.c > 0 ) ?
373  (2.0 * pow (double (rep->_derivedCounts[i].second.c), 2.0)) / denom : 0. ;
374  H += (rep->_derivedCounts[i].second.t > 0 ) ?
375  (2.0 * pow (double (rep->_derivedCounts[i].second.t), 2.0)) / denom : 0. ;
376  H += (rep->_derivedCounts[i].second.zero > 0) ?
377  (2.0 * pow (double (rep->_derivedCounts[i].second.zero), 2.0)) / denom : 0. ;
378  H += (rep->_derivedCounts[i].second.one > 0 ) ?
379  (2.0 * pow (double (rep->_derivedCounts[i].second.one), 2.0)) / denom : 0. ;
380  }
381  else if (numDer == 2 && rep->_haveOutgroup) //MUST have outgroup--else can't proceed
382  { //use a "missing data" scheme if there is >1 derived state
383  //iterate over derived states
384  unsigned config[2];
385  unsigned k = 0;
386  if(rep->_derivedCounts[i].second.a > 0)
387  {
388  config[k++] = rep->_derivedCounts[i].second.a;
389  }
390  if(rep->_derivedCounts[i].second.g > 0)
391  {
392  config[k++] = rep->_derivedCounts[i].second.g;
393  }
394  if(rep->_derivedCounts[i].second.c > 0)
395  {
396  config[k++] = rep->_derivedCounts[i].second.c;
397  }
398  if(rep->_derivedCounts[i].second.t > 0)
399  {
400  config[k++] = rep->_derivedCounts[i].second.t;
401  }
402  if(rep->_derivedCounts[i].second.zero > 0)
403  {
404  config[k++] = rep->_derivedCounts[i].second.zero;
405  }
406  if(rep->_derivedCounts[i].second.one > 0)
407  {
408  config[k++] = rep->_derivedCounts[i].second.one;
409  }
410  for (int i = 0; i < 2; ++i)
411  {
412  double sample_size_adjust = (i==0) ? double(config[1]) : double(config[0]);
413  H += (2.0 *
414  pow (double (config[i]),
415  2.0)) / ((double(samplesize) -
416  sample_size_adjust)
417  * (double(samplesize) -
418  sample_size_adjust
419  - 1.0));
420  }
421  }
422  }
423  }
424  }
425  return H;

674  double PolySNP::Hprime (const bool & likeThorntonAndolfatto) const
687  {
688  assert ( rep->_preprocessed );
689  if(rep->_NumPoly==0) return std::numeric_limits<double>::quiet_NaN();
690  assert(rep->_haveOutgroup==true);
691  double a = a_sub_n ();
692  double b = b_sub_n ();
693  double pi = ThetaPi ();
694  double theta = ThetaW();
695 
696  double thetal = ThetaL();
697  double b_n_plus1 = b_sub_n_plus1();
698  double S = (rep->_totMuts) ? NumMutations() : NumPoly();
699  double thetasq = (likeThorntonAndolfatto == false ) ? S * (S-1)/(a*a + b) : theta*theta;
700 
701  double vThetal =
702  (rep->_totsam * theta)/(2.0 * (rep->_totsam - 1.0))
703  + (2.0 * pow(rep->_totsam/(rep->_totsam - 1.0), 2.0) * (b_n_plus1 - 1.0) - 1.0) * thetasq;
704 
705  double vPi =
706  (3.0 * rep->_totsam *(rep->_totsam + 1.0) * theta
707  + 2.0 * ( rep->_totsam * rep->_totsam + rep->_totsam + 3.0) * thetasq )
708  / (9 * rep->_totsam * (rep->_totsam -1.0));
709 
710  double cov =
711  ((rep->_totsam + 1.0) / (3.0 * (rep->_totsam - 1.0))) * theta
712  +(( 7.0 * rep->_totsam * rep->_totsam + 3.0 * rep->_totsam - 2.0 - 4.0 * rep->_totsam *( rep->_totsam + 1.0) * b_n_plus1)
713  / (2.0 * pow ((rep->_totsam - 1.0), 2.0)))
714  * thetasq ;
715 
716  double Hpr = pi - thetal;
717  Hpr /= pow ( (vThetal + vPi - 2.0 * cov), 0.5);
718  return (Hpr);
719  }

722  double
723  PolySNP::Dnominator (void) const
728  {
729  assert ( rep->_preprocessed );
730  if(rep->_NumPoly==0) return std::numeric_limits<double>::quiet_NaN();
731  double S = 0.0;
732  if (rep->_totMuts)
733  {
734  S = double (NumMutations ());
735  }
736  else if (!(rep->_totMuts))
737  {
738  S = double (NumPoly ());
739  }
740  double a1, a2, b1, b2, c1, c2, e1, e2;
741 
742  a1 = a_sub_n ();
743  a2 = b_sub_n ();
744  b1 = (rep->_totsam + 1.0) / (3.0 * (rep->_totsam - 1.0));
745  b2 = (2.0 * (pow (rep->_totsam, 2.0)
746  + rep->_totsam + 3.0)) / (9.0 * rep->_totsam *
747  (rep->_totsam - 1.0));
748  c1 = b1 - 1.0 / a1;
749  c2 = b2 - (rep->_totsam + 2.0) / (a1 * rep->_totsam) + a2 / pow (a1, 2.0);
750  e1 = c1 / a1;
751  e2 = c2 / (pow (a1, 2.0) + a2);
752  double denominator = pow ((e1 * S + e2 * S * (S - 1.0)), 0.5);
753  return (denominator);
754  }
