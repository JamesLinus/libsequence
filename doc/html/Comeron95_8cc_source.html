203  double Q0_site = q0 / sites->L0();

205  double Q4_site = q4 / sites->L4();
206 
207  log1 = log (1.0 - (2.0 * P2S_site) - Qa);
208  log2 = log (1.0 - (2.0 * Qa));
209 
210  if (!isfinite (log1)) //set value to 0 if nan
211  log1 = 0.0;
212  if (!isfinite (log2))
213  log2 = 0.0;
214 
215  A2S = (-0.5) * log1 + (0.25) * log2;
216 
217  log1 = log (1.0 - (2.0 * P4_site) - Q4_site);
218  log2 = log (1.0 - (2.0 * Q4_site));
219 
220  if (!isfinite (log1))
221  log1 = 0.0;
222  if (!isfinite (log2))

229

235  if (!isfinite (log2))

239 
240  log1 = log (1.0 - (2.0 * P0_site) - Q0_site);
241  log2 = log (1.0 - (2.0 * Q0_site));
242  if (!isfinite (log1))

244  if (!isfinite (log2))
245  log2 = 0.0;
246 
247  A0 = (-0.5) * log1 + (0.25) * log2;

250  sites->L0());
251 
252  if (As <= 0.0)
253  As = 0.0;
254  if (Bs <= 0.0)
255  Bs = 0.0;
256  if (Aa <= 0.0)
257  Aa = 0.0;
