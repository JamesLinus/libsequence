88  bool PolyTable::assign(PolyTable::const_site_iterator beg,
89  PolyTable::const_site_iterator end)
90 
100  {
101  non_const_access = true; //set to true in case an exception is thrown
102  first.clear();
103  second.clear();
104  if(std::distance(beg,end) < 1) return true;
105  first.resize(std::vector<double>::size_type(end-beg));
106  pv.resize(std::vector<double>::size_type(end-beg));
107  second.resize(beg->second.length());
108  size_t nsam = beg->second.length();
109  std::string::const_iterator sb,se;
110  typedef PolyTable::const_site_iterator::difference_type DTYPE;
111  DTYPE i=0,j=0;
112  while((beg+i)<end)
113  {
114  pv[unsigned(i)]=*(beg+i);
115  if ((beg+i)->second.length() != nsam)
116  {
117  //If we toss an exception, let's make sure we leave an empty object.
118  first.clear();
119  second.clear();
120  return false;
121  }
122  first[unsigned(i)] = (beg+i)->first;
123  sb = (beg+i)->second.begin();
124  se = (beg+i)->second.end();
125  j = 0;
126  while( (sb+j) < se )
127  {
128  second[unsigned(j)] += *(sb+j);
129  ++j;
130  }
131  ++i;
132  }
133  non_const_access = false; //everything worked, all private data are assigned, so set to false
134  return true;

352  void PolyTable::ApplyFreqFilter(const unsigned & mincount,
353  const bool & haveOutgroup,
354  const unsigned & outgroup)
366  {
367  std::vector<double> newpos;
368  std::vector<std::string> newdata(second.size());
369 
370  for(unsigned i = 0 ; i < first.size() ; ++i)
371  {
372  stateCounter Counts;
373  for (unsigned j = 0 ; j < second.size() ; ++j)
374  {
375  if(haveOutgroup==false||
376  (haveOutgroup==true && j != outgroup))
377  {
378  Counts(second[j][i]);
379  }
380  }
381  bool freq = true;
382  if (Counts.a > 0 && Counts.a < mincount)
383  {
384  freq = false;
385  }
386  else if (freq == true && Counts.g > 0 && Counts.g < mincount)
387  {
388  freq = false;
389  }
390  else if (freq == true && Counts.c > 0 && Counts.c < mincount)
391  {
392  freq = false;
393  }
394  else if (freq == true && Counts.t > 0 && Counts.t < mincount)
395  {
396  freq = false;
397  }
398  else if (freq == true && Counts.zero > 0 && Counts.zero < mincount)
399  {
400  freq = false;
401  }
402  else if (freq == true && Counts.one > 0 && Counts.one < mincount)
403  {
404  freq = false;
405  }
406  if(freq == true)
407  {
408  newpos.push_back(first[i]);
409  for(unsigned j = 0 ; j < second.size() ; ++j)
410  {
411  newdata[j] += second[j][i];
412  }
413  }
414  }
415  //take care of case where new data are empty
416  if( newpos.empty()) newdata.clear();
417  //Assign takes care of setting non_const_access = true
418  assign(std::move(newpos),std::move(newdata));

542  PolyTable::Binary (const bool & haveOutgroup,
543  const unsigned & outgroup,
544  const bool & strictInfSites )
557  {
558  unsigned nsites = this->numsites();
559 
560  std::vector<double> newpositions;
561  std::vector<unsigned> _pos_indexes;
562  std::vector<std::string> newdata((*this).size());
563 
564  std::string _outgroup;
565  if(haveOutgroup == true)
566  {
567  _outgroup = (*this)[outgroup];
568  }
569  else
570  {
571  _outgroup = (*this)[0];
572  }
573 
574  for(unsigned j =0 ; j < nsites ; ++j)
575  {
576  stateCounter Counts;
577  for(unsigned i = 0 ; i < (*this).size() ; ++i)
578  {
579  Counts ((*this)[i][j]);
580  }
581  if(Counts.nStates()==2)
582  {
583  newpositions.push_back(this->position(j));
584  _pos_indexes.push_back(j);
585  }
586  }
587 
588  for(unsigned i = 0 ; i < (*this).size() ; ++i)
589  {
590  for(unsigned j = 0 ; j < _pos_indexes.size() ; ++j)
591  {
592  if(haveOutgroup==false ||
593  (haveOutgroup==true && _outgroup[j] != 'N'))
594  //skip sites where the ancestral information
595  //is missing AND an outgroup sequence is present
596  {
597  if ( (*this)[i][_pos_indexes[j]] == 'N')
598  {
599  //missing data (leave untouched...)
600  newdata[i] += 'N';
601  }
602  else if( (*this)[i][_pos_indexes[j]]
603  != _outgroup[_pos_indexes[j]] )
604  {
605  //derived (maybe...)
606  newdata[i] += '1';
607  }
608  else if ( (*this)[i][_pos_indexes[j]]
609  == _outgroup[_pos_indexes[j]] )
610  {
611  //ancestral
612  newdata[i] += '0';
613  }
614 
615  }
616  }
617  }
618  //assign takes care of setting non_const_access = true
619  assign(std::move(newpositions),std::move(newdata));
620  }
