95  PolySites::fillIt (const std::vector < __DataType >&alignment, bool strict, bool ignoregaps,
96  bool skipMissing,unsigned freqfilter)
97  {
98  static_assert( std::is_same<__DataType,std::string>::value ||
99  std::is_base_of<Sequence::Seq,__DataType>::value,
100  "__DataType must be std::string or derived from Sequence::Seq");
101  decltype(alignment.size()) numseqs = alignment.size();
102  if(!numseqs) return;
103  std::vector<double> _positions;
104  decltype(alignment[0].length()) seqlen = alignment[0].length();
105  for(decltype(seqlen) j = 0 ; j < seqlen ; ++j)
106  {
107  stateCounter Counts;
108  bool haveMissing = false;
109  for(unsigned i = 0 ; i < numseqs ; ++i)
110  {
111  if (std::toupper(alignment[i][j]) == 'N'||
112  (alignment[i][j]==IDENTICAL && std::toupper(alignment[0][j]) == 'N'))
113  {
114  haveMissing = true;
115  }
116  if (alignment[i][j] == IDENTICAL)
117  {
118  Counts(alignment[0][j]);
119  }
120  else
121  {
122  Counts(alignment[i][j]);
123  }
124  }
125  if ( (ignoregaps == true && Counts.gap == 0) || ignoregaps == false )//skip gaps if necessary
126  {
127  if (skipMissing == false || (skipMissing == true && haveMissing == false))
128  {
129  unsigned numstates = Counts.nStates();
130  unsigned minor_count = SEQMAXUNSIGNED;
131  if ( freqfilter > 0 )
132  {
133  minor_count = (Counts.a > 0 && (Counts.a < minor_count ||
134  minor_count == SEQMAXUNSIGNED))
135  ? Counts.a : minor_count;
136  minor_count = (Counts.g > 0 && (Counts.g < minor_count ||
137  minor_count == SEQMAXUNSIGNED))
138  ? Counts.g : minor_count;
139  minor_count = (Counts.c > 0 && (Counts.c < minor_count ||
140  minor_count == SEQMAXUNSIGNED))
141  ? Counts.c : minor_count;
142  minor_count = (Counts.t > 0 && (Counts.t < minor_count ||
143  minor_count == SEQMAXUNSIGNED))
144  ? Counts.t : minor_count;
145  minor_count = (Counts.zero > 0 && (Counts.zero < minor_count ||
146  minor_count == SEQMAXUNSIGNED))
147  ? Counts.zero : minor_count;
148  minor_count = (Counts.one > 0 && (Counts.one < minor_count ||
149  minor_count == SEQMAXUNSIGNED))
150  ? Counts.one : minor_count;
151  }
152  else if (freqfilter == 0 )
153  minor_count = 1;
154 
155  if(strict == false && numstates > 1 && minor_count > freqfilter)
156  //site is variable
157  {
158  _positions.push_back(double(j+1));//add 1 to emulate a real positions
159  }
160  else if(strict == true && numstates == 2 && minor_count > freqfilter )
161  //if there are 2 and only 2 character states
162  {
163  _positions.push_back(double(j+1));//add 1 to emulate a real positions
164  }
165  }
166  }
167  }
168  std::vector<std::string> _data(numseqs);
169  for(unsigned i = 0 ; i < numseqs ; ++i)
170  {
171  for(unsigned j = 0 ; j < _positions.size() ; ++j)
172  {
173  if (alignment[i][unsigned(_positions[j])-1] == IDENTICAL)
174  {
175  _data[i] += alignment[0][unsigned(_positions[j])-1];
176  }
177  else
178  {
179  _data[i] += alignment[i][unsigned(_positions[j])-1];
180  }
181  }
182  }
183  PolyTable::assign(&_positions[0],_positions.size(),&_data[0],_data.size());
184  }
