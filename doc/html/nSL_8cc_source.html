57  /*
58  Mechanics of the nSL statistic
59 
60  RV = nSL,iHS, as defined in doi:10.1093/molbev/msu077
61  */
62  std::array<double, 4>
63  __nSLdetails(const std::size_t &core, const Sequence::SimData &d,
64  // const vector<size_t> &coretype,
65  const std::unordered_map<double, double> &gmap)
66  {
67  auto csize = d.size();
68  // This tracks s,s2 for ancestral and derived
69  // mutation, resp:
70  std::array<double, 4> rv = { 0., 0., 0., 0. };
71  // number of comparisons for ancestral and
72  // derived, resp:
73  std::array<unsigned, 2> nc = { 0u, 0u };
74  for (size_t i = 0; i < csize; ++i)
75  {
76  auto start = d[i].cbegin();
77  auto bi
78  = start
79  + static_cast<decltype(start)::difference_type>(core);
80  size_t iIsDer = (*bi == '1');
81  for (size_t j = i + 1; j < csize; ++j)
82  {
83  auto bj
84  = d[j].cbegin()
85  + static_cast<decltype(start)::difference_type>(
86  core);
87  size_t jIsDer = (*bj == '1');
88  if (iIsDer == jIsDer)
89  {
90  auto eri = d[i].crend();
91  auto ei = d[i].cend();
92  auto right = mismatch(bi, ei, bj);
93  string::const_reverse_iterator ri1(bi),
94  ri2(bj);
95  auto left = mismatch(ri1, eri, ri2);
96  if (left.first != eri && right.first != ei)
97  {
98  rv[2 * iIsDer] += static_cast<double>(
99  distance(left.first.base(),
100  right.first)
101  + 1);
102  rv[2 * iIsDer + 1] += update_s2(
103  start, left.first.base(),
104  right.first, d, gmap);
105  nc[iIsDer]++;
106  }
107  }
108  }
109  }
110  rv[0] /= static_cast<double>(nc[0]);
111  rv[1] /= static_cast<double>(nc[0]);
112  rv[2] /= static_cast<double>(nc[1]);
113  rv[3] /= static_cast<double>(nc[1]);
114  return rv;
115  }

132  nSL_t(const SimData &d, const std::unordered_map<double, double> &gmap)
133  {
134  vector<pair<double, double>> rv(d.numsites());
135  tbb::parallel_for(
136  tbb::blocked_range<std::size_t>(0, rv.size()),
137  [&rv, &d, &gmap](const tbb::blocked_range<std::size_t> &r) {
138  for (std::size_t i = r.begin(); i < r.end(); ++i)
139  {
140  auto temp = __nSLdetails(i, d, gmap);
141  rv[i] = make_pair(log(temp[0]) - log(temp[2]),
142  log(temp[1]) - log(temp[3]));
143  }
144  });
145  return rv;

148  Return max. abs value of standardized nSL and iHS, with the latter as
149  defined by Ferrer-Admetella et al.
150  */
151  pair<double, double>
152  snSL(const SimData &d, const double minfreq, const double binsize,
153  const std::unordered_map<double, double> &gmap)
154  {
155  if (d.empty())
156  return make_pair(std::numeric_limits<double>::quiet_NaN(),
157  std::numeric_limits<double>::quiet_NaN());
158  vector<polymorphicSite> filtered;
159  vector<unsigned> dcounts;
160  dcounts.reserve(d.numsites());
161  for_each(d.sbegin(), d.send(), [&](const polymorphicSite &p) {
162  unsigned dcount = static_cast<unsigned>(
163  count(p.second.begin(), p.second.end(), '1'));
164  if (dcount && dcount < d.size())
165  {
166  double f = static_cast<double>(dcount)
167  / static_cast<double>(d.size());
168  if (min(f, 1. - f) >= minfreq)
169  {
170  filtered.push_back(p);
171  dcounts.push_back(dcount);
172  }
173  }
174  });
175  if (filtered.empty())
176  return make_pair(std::numeric_limits<double>::quiet_NaN(),
177  std::numeric_limits<double>::quiet_NaN());
178  SimData __filtered(filtered.begin(), filtered.end());
179  // Get the stats
180  auto nSLstats = nSL_t(__filtered, gmap);
181  // Associate the stats with their DAFs
182  using pp = pair<double, pair<double, double>>;
183  vector<pp> binning;
184  for (unsigned i = 0; i < __filtered.numsites(); ++i)
185  {
186  // pair<double, double> rvi = nSL(i, __filtered, gmap);
187  binning.push_back(
188  make_pair(static_cast<double>(dcounts[i])
189  / static_cast<double>(__filtered.size()),
190  nSLstats[i]));
191  }
192  // sort based on DAF
193  std::sort(binning.begin(), binning.end(),
194  [](const pp &a, const pp &b) { return a.first < b.first; });
195  double rv = std::numeric_limits<double>::quiet_NaN(),
196  rv2 = std::numeric_limits<double>::quiet_NaN();
197  // Now, bin, standardise, and move on...
198  vector<pp> thisbin;
199  auto bstart = binning.begin();
200  size_t ttlSNPs = 0;
201  for (double l = minfreq; l < 1.; l += binsize)
202  {
203  thisbin.clear();
204  auto first = lower_bound(
205  bstart, binning.end(), l,
206  [](const pp &p, const double d) { return p.first < d; });
207  auto last = upper_bound(
208  first, binning.end(), l + binsize,
209  [](const double d, const pp &p) { return d <= p.first; });
210  thisbin.insert(thisbin.end(), first, last);
211  ttlSNPs += thisbin.size();
212  bstart = last;
213  if (thisbin.size() > 1) // otherwise SD = 0, so there's
214  // nothing to standardize
215  {
216  double sum1 = 0., sum2 = 0., sumsq1 = 0., sumsq2 = 0.;
217  for (const auto &p : thisbin)
218  {
219  if (isfinite(p.second.first))
220  {
221  sum1 += p.second.first;
222  sumsq1 += pow(p.second.first, 2.);
223  }
224  if (isfinite(p.second.second))
225  {
226  sum2 += p.second.second;
227  sumsq2 += pow(p.second.second, 2.);
228  }
229  }
230  double mean1
231  = sum1 / static_cast<double>(thisbin.size());
232  double mean2
233  = sum2 / static_cast<double>(thisbin.size());
234  double C = static_cast<double>(thisbin.size())
235  / static_cast<double>(thisbin.size() - 1);
236  double var1
237  = C * (sumsq1 / static_cast<double>(thisbin.size())
238  - pow(mean1, 2.));
239  double var2
240  = C * (sumsq2 / static_cast<double>(thisbin.size())
241  - pow(mean2, 2.));
242  double sd1 = sqrt(var1), sd2 = sqrt(var2);
243  for (const auto &data : thisbin)
244  {
245  rv = maxabs(data.second.first, mean1, sd1, rv);
246  rv2 = maxabs(data.second.second, mean2, sd2,
247  rv2);
248  }
249  }
250  }
251  if (ttlSNPs != __filtered.numsites())
252  {
253  throw runtime_error("Incorrect number of SNPs processed");
254  }
255  return make_pair(rv, rv2);
256  }
