358  std::ostream &
359  newick_stream_marginal_tree_impl::parens( const int & noden,
360  std::ostream & o) const
364  {
365  if( left[std::vector<int>::size_type(noden)] == -1 )
366  {
367  assert( (mi+((mi+noden)->abv))->time >= 0. );
368  o << noden+1 << ':' <<(mi+((mi+noden)->abv))->time;
369  }
370  else
371  {
372  o << '(';
373  parens( left[std::vector<int>::size_type(noden)], o ) ;
374  o << ',';
375  parens( right[std::vector<int>::size_type(noden)], o ) ;
376  if( (mi+noden)->abv == -1 )
377  {
378  o << ");";
379  }
380  else
381  {
382  double time = (mi + (mi+noden)->abv )->time - (mi+noden)->time ;
383  assert(time >= 0.);
384  o << "):"<<time;
385  }
386  }
387  return o;
388  }
