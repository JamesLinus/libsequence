102  void
103  RedundancyCom95impl::FillFirstPositionCounts ()
104  {
105  int i, j, k, l;
106  std::string codon,mutation;
107  int numN, numTsS, numTvS; //numbers of changes of various types, N = nonsynon, S = synon
108  int codonState;
109  int type; //type refers to Ts or Tv
110  int numPossTs, numPossTv;
111  double FirstN, First2S, First2V;
112  std::string codonTrans;
113  std::string mutationTrans;
114 
115  bool S, N;
116  //make every codon
117  codon.resize(3);
118  mutation.resize(3);
119  for (i = Nucleotides (A); i <= Nucleotides (C); ++i)
120  {
121  for (j = Nucleotides (A); j <= Nucleotides (C); ++j)
122  {
123  for (k = Nucleotides (A); k <= Nucleotides (C); ++k)
124  {
125  numN = numTsS = numTvS = 0; //initialize
126  numPossTs = numPossTv = 0;
127  S = N = 0;
128  codon[0] = intToNuc[i];
129  codon[1] = intToNuc[j];
130  codon[2] = intToNuc[k];
131 
132  //now make all possible mutations at first position
133  codonState = i;
134  for (l = Nucleotides (A);
135  l <= Nucleotides (C); ++l)
136  {
137  if (l != codonState)
138  { //only include mutations
139  mutation[0] = intToNuc[l];
140  mutation[1] = codon[1];
141  mutation[2] = codon[2]; //the mutant codon is now defined
142  type = TsTv (intToNuc[codonState], mutation[0]);//find out if change is Ts or Tv
143 
144  codonTrans = Translate (codon.begin(),
145  codon.end(), genetic_code);
146  mutationTrans = Translate (mutation.begin(),
147  mutation.end(),genetic_code);
148 
149  //if neither codon is a stop
150  if ((codonTrans)[0] != '*'
151  && (mutationTrans)[0] != '*')
152  {
153  if(codonTrans == mutationTrans)
154  { //synonymous change
155  S = 1;
156  N = 0;
157  }
158  else
159  { //nonsynonymous change
160  S = 0;
161  N = 1;
162  }
163  if (type == Mutations(Ts))
164  ++numPossTs;
165  else if (type == Mutations(Tv))
166  ++numPossTv;
167  if (N == 1) //if change is nonsynonymous
168  ++numN;
169  else if (S == 1 && type == Mutations(Ts))
170  //if change is synonymous transition
171  ++numTsS;
172  else if (S == 1 && type == Mutations(Tv))
173  //if change is synonymous transversion
174  ++numTvS;
175  }
176  }
177  }
178  //process here
179  if ((numPossTs + numPossTv) == 0)
180  { //stop codons
181  First2S = 0;
182  First2V = 0;
183  FirstN = 0;
184  }
185  else if ((numTsS + numTvS) == 0) //non-degenerate sites
186  {
187  First2S = 0;
188  First2V = 0;
189  FirstN = 1.0;
190  }
191  else if (numPossTs != 0 && numPossTv != 0)
192  { //special cases
193  if (double (numTsS) /
194  double (numPossTs) != 1.0
195  && double (numTvS) /
196  double (numPossTv) != 1.0)
197  { //fractional redundancy
198  First2S =
199  double (numTsS) /
200  double (numPossTs +
201  numPossTv);
202  First2V =
203  double (numTvS) /
204  double (numPossTs +
205  numPossTv);
206  FirstN = 1.0 - First2S -
207  First2V;
208  }
209  else
210  { //odd types of degeneracy
211  First2S =
212  double (numTsS) /
213  double (numPossTs);
214  First2V =
215  double (numTvS) /
216  double (numPossTv);
217  FirstN = 1.0 - First2S -
218  First2V;
219  }
220  }
221  else
222  {
223  if (numPossTs > 0)
224  First2S =
225  double (numTsS) /
226  double (numPossTs + numPossTv);
227  else
228  First2S = 0.0;
229 
230  if (numPossTv > 0)
231  First2V =
232  double (numTvS) /
233  double (numPossTv + numPossTs);
234  else
235  First2V = 0.0;
236  FirstN = 1.0 - First2S - First2V;
237  }
238  //fill array
239  firstNon[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=FirstN;
240  first2S[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=First2S;
241  first2V[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=First2V;
242  }
243  }
244  }

248  RedundancyCom95impl::FillThirdPositionCounts ()
249  {
250  int i, j, k, l;
251  std::string codon, mutation; //the two codons
252  int numN, numTsS, numTvS; //numbers of changes of various types, N = nonsynon, S = synon
253  int codonState;
254  int type; //type refers to Ts or Tv
255  int numPossTs, numPossTv;
256  double ThirdN, Third2S, Third2V, Third4;
257  std::string codonTrans;
258  std::string mutationTrans;
259  bool S, N;
260  //make every codon
261  codon.resize(3);
262  mutation.resize(3);
263  for (i = Nucleotides (A); i <= Nucleotides (C); ++i)
264  {
265  for (j = Nucleotides (A); j <= Nucleotides (C); ++j)
266  {
267  for (k = Nucleotides (A); k <= Nucleotides (C); ++k)
268  {
269  numN = numTsS = numTvS = 0; //initialize
270  numPossTs = numPossTv = 0;
271  S = N = 0;
272  codon[0] = intToNuc[i];
273  codon[1] = intToNuc[j];
274  codon[2] = intToNuc[k];
275  //now make all possible mutations at first position
276  codonState = k;
277 
278  for (l = Nucleotides (A);
279  l <= Nucleotides (C); ++l)
280  {
281  if (l != codonState)
282  { //only include mutations
283  mutation[0] = codon[0];
284  mutation[1] = codon[1];
285  mutation[2] = intToNuc[l]; //the mutant codon is now defined
286  type = TsTv (intToNuc[codonState], mutation[2]); //find out if change is Ts or Tv
287 
288  codonTrans = Translate (codon.begin(),
289  codon.end(),genetic_code);
290  mutationTrans = Translate (mutation.begin(),
291  mutation.end(),genetic_code);
292 
293  if ((codonTrans)[0] != '*'
294  && (mutationTrans)[0] != '*')
295  {
296  if(codonTrans==mutationTrans)
297  {
298  S = 1;
299  N = 0;
300  }
301  else
302  {
303  S = 0;
304  N = 1;
305  }
306  if (type == Mutations(Ts))
307  ++numPossTs;
308  else if (type == Mutations(Tv))
309  ++numPossTv;
310  if (N)
311  ++numN;
312  else if (S && type == Mutations(Ts))
313  ++numTsS;
314  else if (S && type == Mutations(Tv))
315  ++numTvS;
316  }
317  }
318  }
319  //process here
320  ThirdN = Third2S = Third2V = Third4 = 0.0;
321  if (numPossTs + numPossTv == 0)
322  { //stop codons
323  ThirdN = 0.0;
324  Third2S = 0.0;
325  Third2V = 0.0;
326  Third4 = 0.0;
327  }
328  else if (numTsS + numTvS == 0)
329  { //nondegenerate
330  ThirdN = 1.0;
331  Third2S = 0.0;
332  Third2V = 0.0;
333  Third4 = 0.0;
334  }
335  else if (numTsS + numTvS == 3)
336  { //4fold degenerate
337  ThirdN = 0.0;
338  Third2S = 0.0;
339  Third2V = 0.0;
340  Third4 = 1.0;
341  }
342  else if (numTsS == 0 || numTvS == 0)
343  { //if there is non-degeneracy for one type
344  if (numPossTs > 0)
345  Third2S =
346  double (numTsS) /
347  double (numPossTs);
348  else
349  Third2S = 0.0;
350  if (numPossTv > 0)
351  Third2V =
352  double (numTvS) /
353  double (numPossTv);
354  else
355  Third2V = 0.0;
356  ThirdN = 1.0 - Third2S - Third2V;
357  Third4 = 0.0;
358  }
359  else if (numTsS > 0 && numTvS > 0
360  && (double (numTsS) /
361  double (numPossTs) != 1.0
362  || double (numTvS) /
363  double (numPossTv) != 1.0))
364  { //fractional degeneracy
365  Third2S = 1.0 / 3.0;
366  Third2V = 1.0 / 3.0;
367  ThirdN = 1.0 / 3.0;
368  Third4 = 0.0;
369  }
370  thirdNon[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=ThirdN;
371  third2S[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=Third2S;
372  third2V[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=Third2V;
373  thirdFour[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]=Third4;
374  }
375  }
376  }

380  void
381  RedundancyCom95impl::FillLValues ()
382  {
383  int i, j, k;
384  std::string trans;
385 
386  std::string codon(3,'A');
387  for (i = Nucleotides (A); i <= Nucleotides (C); ++i)
388  for (j = Nucleotides (A); j <= Nucleotides (C); ++j)
389  for (k = Nucleotides (A); k <= Nucleotides (C); ++k)
390  {
391  codon[0] = intToNuc[i];
392  codon[1] = intToNuc[j];
393  codon[2] = intToNuc[k];
394 
395  trans =Translate (codon.begin(),codon.end(), genetic_code);
396  if(strcmp(trans.c_str(),"*")!=0)
397  { //if not a stop codon
398  l0_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] = 1.
399  + firstNon[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]]
400  + thirdNon[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]];
401 
402  l2S_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] =
403  first2S[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] +
404  third2S[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]];
405 
406  l2V_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] =
407  first2V[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] +
408  third2V[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]];
409 
410  l4_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] =
411  thirdFour[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]];
412  }
413  else
414  {
415  l0_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] = 0.0;
416  l2S_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] = 0.0;
417  l2V_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] = 0.0;
418  l4_vals[nucToInt[codon[0]]][nucToInt[codon[1]]][nucToInt[codon[2]]] = 0.0;
419  }
420  }
